</Tab>

### Manual

<Steps>
  <Step>
    <strong>Install the following dependencies</strong>

    <Tabs items={['npm', 'pnpm','yarn']}>
      <Tab>
        ```sh
        npm install clsx tailwind-merge
        ```
      </Tab>
      <Tab>
        ```sh
        pnpm add clsx tailwind-merge
        ```
      </Tab>
      <Tab>
        ```sh
        yarn add clsx tailwind-merge
        ```
      </Tab>
    </Tabs>
  </Step>

  <Step>
    <strong>If you don't have Tailwind configured yet, add it (dev deps)</strong>

    <Tabs items={['npm', 'pnpm','yarn']}>
      <Tab>
        ```sh
        npm install -D tailwindcss postcss autoprefixer
        npx tailwindcss init -p
        ```
      </Tab>
      <Tab>
        ```sh
        pnpm add -D tailwindcss postcss autoprefixer
        pnpm dlx tailwindcss init -p
        ```
      </Tab>
      <Tab>
        ```sh
        yarn add -D tailwindcss postcss autoprefixer
        yarn dlx tailwindcss init -p
        ```
      </Tab>
    </Tabs>
  </Step>

  <Step>
    <strong>Add a classname utility function</strong>

    ```tsx
    import { clsx, type ClassValue } from 'clsx';
    import { twMerge } from 'tailwind-merge';

    export const cn = (...inputs: ClassValue[]) => {
      return twMerge(clsx(inputs));
    };
    ```
  </Step>

  <Step>
    <strong>Copy and paste the component</strong>

    <ComponentSource name='bounce-button' />
  </Step>

  <Step>
    <strong>Update the import paths</strong>

    <p>Adjust import paths to match your project's folder structure.</p>
  </Step>

</Steps>
